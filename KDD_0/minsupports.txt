inputs = db, minsup [0-1]

frequent pattern = has frequency >= minsup

a = number of transactions that has the sequence
b = total number of sequences = 492

minsup = a / 492

# Minimum Support for frequences that appear in more or equals that 50% off all transactions

492 / 2 = 246
minsup = 246 / 492
minsup = 0.5

# Minimum Support for frequences that appear in more or equals that 75% off all transactions

(492 / 4) * 3 = 369
minsup = 369 / 492
minsup = 0.75

# Minimum Support for frequences that appear in  100% off all transactions

minsup = 492 / 492
minsup = 1
